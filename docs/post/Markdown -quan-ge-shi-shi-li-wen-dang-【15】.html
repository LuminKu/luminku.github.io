4. 代码与语法高亮

行内代码

代码块（JavaScript）

// 斐波那契数列生成函数
function fibonacci(n) {
  if (n <= 1) return n;
  let a = 0, b = 1;
  for (let i = 2; i <= n; i++) {
    [a, b] = [b, a + b];
  }
  return b;
}

// 使用示例
console.log(fibonacci(10)); // 输出: 55

代码块（Python）

def factorial(n):
    """计算阶乘"""
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)

# 列表推导式示例
squares = [x**2 for x in range(10)]
print(squares)  # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]
